-- Animate Smiley (Pulse Effect)
local function pulseEmoji()
	local scaleUp = TweenService:Create(SmileyLabel, TweenInfo.new(0.5, Enum.EasingStyle.Sine), {TextSize = 34})
	local scaleDown = TweenService:Create(SmileyLabel, TweenInfo.new(0.5, Enum.EasingStyle.Sine), {TextSize = 28})

	while SmileyLabel and SmileyLabel.Parent do
		scaleUp:Play()
		scaleUp.Completed:Wait()
		scaleDown:Play()
		scaleDown.Completed:Wait()
	end
end
task.spawn(pulseEmoji)
