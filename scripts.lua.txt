-- Function to create pulse animation
local function createPulse(label)
	local up = TweenService:Create(label, TweenInfo.new(0.5, Enum.EasingStyle.Sine), {TextSize = 34})
	local down = TweenService:Create(label, TweenInfo.new(0.5, Enum.EasingStyle.Sine), {TextSize = 28})
	task.spawn(function()
		while label and label.Parent do
			up:Play()
			up.Completed:Wait()
			down:Play()
			down.Completed:Wait()
		end
	end)
end

end
