-- Animate the Smiley Face (pulse)
local function pulseEmoji()
	local scaleUp = TweenService:Create(SmileyLabel, TweenInfo.new(0.5, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {TextSize = 34})
	local scaleDown = TweenService:Create(SmileyLabel, TweenInfo.new(0.5, Enum.EasingStyle.Sine, Enum.EasingDirection.In), {TextSize = 28})

	while SmileyLabel and SmileyLabel.Parent do
		scaleUp:Play()
		scaleUp.Completed:Wait()
		scaleDown:Play()
		scaleDown.Completed:Wait()
	end
end

task.spawn(pulseEmoji)
